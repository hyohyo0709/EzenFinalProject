1. 테이블생성 
create table ezenboards(
   	num number,
   	member_id varchar2(20),     
 	member_email varchar2(30),
	subject varchar2(50),
	reg_date date,
	readcount number default 0, 
	ref number, 
	re_step number, 
	re_level number,
	board_type number, 
	content varchar2(100),
    upload varchar2(300)
);

member_id, member_email = 로그인시에 받아올값

2.게시판 번호 자동생성

board_type  0:자유 게시판
board_type  1:상품리뷰 게시판
board_type  2:상품문의 게시판
